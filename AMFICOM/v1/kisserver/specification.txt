	Конструкторы.

public Test(ClientTest_Transferable tt) throws SQLException
	Создать и сохранить в БД новый тест.

public Test(String id) throws SQLException
	Восстановить из БД тест с идентификатором id.


	Методы.

public void delete() throws SQLException
	Отметить данный тест в БД как удалённый.

public Analysis getAnalysis()
	Получить объект анализа для данного теста. Если данный тест проводится без анализа, возвращается null.

public ClientTest_Transferable getClientTransferable() throws SQLException
	Получить клиентскую CORBA-структуру.

public Timestamp getEndTime()
	Получить время окончания теста (имеет смысл только для периодического временного типа).

public Evaluation getEvaluation()
	Получить объект оценки для данного теста. Если данный тест проводится без оценки, возвращается null.

public String getId()
	Получить идентификатор данного теста.

public String getKISId()
	Получить идентификатор КИС данного теста.

public Timestamp getModified()
	Получить время последнего изменения данного теста.

public String getName()
	Получить имя данного теста.

public Timestamp getPeriod()
	Получить период измерений данного теста (имеет смысл только для периодического временного типа).

public int getStatus()
	Получить статус данного теста.

public int getTemporalType()
	Получить временной тип данного теста.

public Timestamp[] getTI()
	Получить временные метки данного теста (имеет смысл только для временного типа "по расписанию").

public Test_Transferable getTransferable() throws SQLException
	Получить агентскую CORBA-структуру.

public String getTypeId()
	Получить идентификатор типа данного теста.

public boolean isModified() throws SQLException
	Истинно, если запись в БД, соответствующая данному тесту, обновлялась позже, чем данный объект 
	прочитывался из БД в последний раз. Т. е., сравниваются значение поля modified в таблице БД и значение
	поля modified данного объекта, и если первое позже последнего - возвращается true.

public static Test[] retrieveAbortedTests(String agent_id) throws SQLException
	Восстановить из БД все тесты, прерванные пользователем, для всех КИСов, которые обслуживаются агентом
	agent_id. Это тесты, у которых статус SCHEDULED и deleted не NULL.

public static Test[] retrieveScheduledTests(String agent_id, long interval) throws SQLException
	Восстановить из БД все тесты, у которых статус SCHEDULED, КИС обслуживается агентом agent_id, а время 
	начала измерений лежит в интервале [SYSDATE, SYSDATE + interval].

public static Test retrieveTestForAnalysis(String analysis_id) throws SQLException
	Восстановить из БД тест, который проводился с анализом analysis_id.

public static Test retrieveTestForEvaluation(String evaluation_id) throws SQLException
	Восстановить из БД тест, который проводился с оценкой evaluation_id.

public static Test[] retrieveTests() throws SQLException
	Восстановить из БД все тесты, не помеченные, как удалённые.

public static Test[] retrieveTests(String domain_id) throws SQLException
	Восстановить из БД все тесты для данного домена, не помеченные, как удалённые.

public void update(ClientTest_Transferable tt) throws SQLException
	Обновить даный тест. В действительности, обновляются только временные характеристики теста - 
	start_time, period, end_time и ti (нужное - в зависимости от временного типа).

		EventSource


	Конструкторы.

public EventSource(String id) throws SQLException
	Восстановить из БД источник событий по идентификатору. Для типа данного источника восстанавливаются
	типы событий, которые на нём могут возникать, и для каждого них устанавливается правило обработки,

	какого-либо типа можно с помощью метода setRule данного класса.

public EventSource(String type_id,
                   String object_source_id,
                   String description) throws SQLException
